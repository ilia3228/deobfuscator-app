  return { high: C.red, med: C.orange, low: C.textDim }[sev] || C.textDim;
}
function sevBg(sev) {
  return { high: C.redDim, med: C.orangeDim, low: '#1e1e1e' }[sev] || '#1e1e1e';

function LogStrip({ running, done, lt }) {
  const lth = lt || getLangTheme(null);
  const [open, setOpen] = useState(true);
  const [filter, setFilter] = useState('all'); // all | info | debug | ok
  const logRef = useRef(null);

  const levelColor = { DEBUG: '#555555', INFO: '#5a8fd4', OK: '#e0e0e0', WARN: C.orange, ERR: C.red };
  const entryColor = (e) => {
    if (e.level === 'OK')   return '#ffffff';
    if (e.level === 'INFO') return '#7aaee8';
    if (e.level === 'ERR')  return C.red;
    if (e.level === 'WARN') return C.orange;
    if (/\d+%/.test(e.text)) return C.orange;
    if (e.text.startsWith('↳')) return '#8aaecc';
    if (/not applicable/.test(e.text)) return '#353535';
    if (/converged|no changes/.test(e.text)) return '#484848';
    return '#909090';
  };

  const visible = filter === 'all'
    ? LOG_ENTRIES
    : LOG_ENTRIES.filter(e => e.level === filter.toUpperCase());

  useEffect(() => {
    if (open && logRef.current) logRef.current.scrollTop = logRef.current.scrollHeight;
  }, [open]);

  const isLayerHeader = e => e.text.startsWith('──');
  const isDone = e => e.level === 'OK' && e.indent === 0;

  return (
    <div style={{ borderTop:`1px solid ${C.border}`, background: C.bg1, flexShrink:0 }}>
      {/* header bar */}
      <div style={{ display:'flex', alignItems:'center', borderBottom: open ? `1px solid ${C.border}` : 'none' }}>

          style={{ background:'none', border:'none', cursor:'pointer',

          <span style={{ fontFamily: C.mono, fontSize:11, color: C.textMuted }}>log</span>
        </button>

          background: C.bg3, border:`1px solid ${C.border}`, color: C.textMuted }}>

        </span>
        {running && (

        )}
        {done && (

// ─── syntax coloriser ─────────────────────────────────────────────────────────
function colorize(line) {
  // syntax highlighting recognises both JS and Python keywords (no theming)
  const isPyComment = line.trimStart().startsWith('#') && !line.includes('://');
  if (isPyComment || line.trimStart().startsWith('//')) return '#7ab568';
  if (/⚠/.test(line)) return '#d46a6a';
  if (/\b(async|await|function|return|const|let|var|if|else|new|this|typeof|def|import|from|class|lambda|None|True|False|self|elif|try|except|with|raise|yield|in|not|and|or|pass)\b/.test(line)) return '#8ab8d8';
  if (/'[^']*'|"[^"]*"|`[^`]*`/.test(line)) return '#cc9e78';
  if (/0x[0-9a-fA-F]+/.test(line)) return '#aabe8a';
  return '#c8c8c0';
}
